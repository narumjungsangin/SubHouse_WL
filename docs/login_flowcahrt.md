graph TD
    A[시작] --> B{랜딩페이지}

    %% 좌측 흐름: 매물 찾기
    B -- 매물 찾기 --> C{추천 집}
    C --> E
    C --검색 필요--> Ca[검색]
    Ca --> D[필터]
    D --> E[로그인 여부]
    E --> Ee{ID 여부}
    Ee -- NO ID --> F[회원가입]
    Ee -- YES ID --> G[계약 내용]
    F --> G
    G --> H[end]

    %% 우측 흐름: 계약 등록
    B -- 매물 등록 --> I[로그인]
    I --> J{ID 여부}
    J -- no ID --> F2[회원가입]
    F2 --> K
    J -- YES ID --> K[계약 등록하기]
    K --> L[아파트 이름]
    L --> M[집 금액, 집 위치, 방개수, 집크기, 가구유무, 사진]
    M --> N[계약자 명, 계약 기간]
    N --> O[정보 등록]
    O --> H
